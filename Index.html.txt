/* ══════════════════════════════════════
   SILHOUETTE SVG GENERATOR
   Level 0–9 → evolving human posture
══════════════════════════════════════ */
function silSVG(level, color='white'){
  const L = Math.min(9, Math.max(0, level));
  // Fixed canvas: 40 wide, 56 tall
  const VW = 40, VH = 56, cx = 20;
  const op = 0.3 + L * 0.07;

  const headY = 7 - L * 0.2;
  const headR = 5 + L * 0.1;

  // Spine bottom Y: rises as posture improves
  const spineBot = 50 - L * 0.5;
  // Spine path — collapsed→upright
  let spine;
  if(L === 0)      spine = `M${cx} ${headY+headR+1} Q${cx-7} ${headY+18} ${cx-5} ${headY+32} Q${cx-3} ${headY+43} ${cx-1} ${spineBot}`;
  else if(L === 1) spine = `M${cx} ${headY+headR+1} Q${cx-4} ${headY+16} ${cx-3} ${headY+30} L${cx-2} ${headY+44} L${cx} ${spineBot}`;
  else if(L === 2) spine = `M${cx} ${headY+headR+1} Q${cx-2} ${headY+14} ${cx-1} ${headY+28} L${cx} ${headY+43} L${cx+1} ${spineBot}`;
  else             spine = `M${cx} ${headY+headR+1} L${cx} ${headY+26} L${cx} ${spineBot}`;

  // Arms: droop at L0, horizontal at L5, rise at L9
  const armAttachY = headY + headR + 7 + Math.max(0,(4-L)*1.5);
  const armSpread  = 5 + L * 2.8;    // max ~30 at L9 — stays within VW
  const armDrop    = 7 - L * 1.4;    // goes negative (arms rise) after L5

  // Legs
  const legTopY = headY + headR + 28 - L * 0.5;
  const legBot  = Math.min(VH - 2, headY + headR + 48);
  const legSpr  = 5 + L * 1.0;

  // Aura rings — small, contained within viewBox
  let aura = '';
  if(L >= 5){
    const aOp = (L-4)*0.09;
    const bodyCy = headY + headR + 20;
    const r1 = Math.min(14 + (L-4)*2, 17);
    const r2 = Math.min(r1 + 5, 21);
    aura = `<circle cx="${cx}" cy="${bodyCy}" r="${r1}" fill="none" stroke="${color}" stroke-width=".8" opacity="${aOp}"/>
            <circle cx="${cx}" cy="${bodyCy}" r="${r2}" fill="none" stroke="${color}" stroke-width=".5" opacity="${aOp*.55}"/>`;
  }

  // Sparks — kept inside VW
  let sparks = '';
  if(L >= 7){
    const cnt = (L-6)*4;
    const sR  = 16 + (L-7)*3;
    for(let i=0;i<cnt;i++){
      const angle = (i/cnt)*Math.PI*2;
      const px = cx + Math.cos(angle)*sR;
      const py = (headY+headR+20) + Math.sin(angle)*sR*0.6;
      if(px>1&&px<VW-1&&py>1&&py<VH-1)
        sparks += `<circle cx="${px.toFixed(1)}" cy="${py.toFixed(1)}" r="1" fill="${color}" opacity="${(0.25+(L-7)*0.1).toFixed(2)}"/>`;
    }
  }

  return `<svg viewBox="0 0 ${VW} ${VH}" xmlns="http://www.w3.org/2000/svg" style="display:block;width:38px;height:52px;flex-shrink:0;overflow:hidden">
    <defs><clipPath id="cb"><rect width="${VW}" height="${VH}"/></clipPath></defs>
    <g clip-path="url(#cb)">
      ${aura}${sparks}
      <circle cx="${cx}" cy="${headY}" r="${headR}" fill="${color}" opacity="${(op+.05).toFixed(2)}"/>
      <path d="${spine}" stroke="${color}" stroke-width="3" fill="none" stroke-linecap="round" opacity="${op.toFixed(2)}"/>

function makeSilEl(level){
  // Create a real DOM SVG element — not innerHTML string
  // This avoids clipPath id collision and Safari inline-SVG sizing bugs
  const NS = 'http://www.w3.org/2000/svg';
  const L  = Math.min(9, Math.max(0, level));
  const VW = 40, VH = 56, cx = 20;
  const op = (0.3 + L * 0.07).toFixed(2);

  const svg = document.createElementNS(NS, 'svg');
  svg.setAttribute('viewBox', `0 0 ${VW} ${VH}`);
  svg.setAttribute('width',  '40');
  svg.setAttribute('height', '56');
  svg.style.cssText = 'display:block;width:40px;height:56px;flex-shrink:0';

  const mk = (tag, attrs) => {
    const el = document.createElementNS(NS, tag);
    for(const [k,v] of Object.entries(attrs)) el.setAttribute(k,v);
    return el;
  };

  const headY = 7 - L * 0.2;
  const headR = 5 + L * 0.1;

  // Aura rings
  if(L >= 5){
    const aOp  = ((L-4)*0.09).toFixed(2);
    const bcy  = headY + headR + 20;
    const r1   = Math.min(14+(L-4)*2, 17);
    const r2   = Math.min(r1+5, 21);
    svg.appendChild(mk('circle',{cx,cy:bcy,r:r1,fill:'none',stroke:'white','stroke-width':'.8',opacity:aOp}));
    svg.appendChild(mk('circle',{cx,cy:bcy,r:r2,fill:'none',stroke:'white','stroke-width':'.5',opacity:((L-4)*0.05).toFixed(2)}));
  }

  // Sparks
  if(L >= 7){
    const cnt = (L-6)*4, sR = 16+(L-7)*3;
    for(let i=0;i<cnt;i++){
      const a  = (i/cnt)*Math.PI*2;
      const px = cx+Math.cos(a)*sR, py = (headY+headR+20)+Math.sin(a)*sR*0.6;
      if(px>1&&px<VW-1&&py>1&&py<VH-1)
        svg.appendChild(mk('circle',{cx:px.toFixed(1),cy:py.toFixed(1),r:'1',fill:'white',opacity:(0.25+(L-7)*0.1).toFixed(2)}));
    }
  }

  // Head
  svg.appendChild(mk('circle',{cx,cy:headY,r:headR,fill:'white',opacity:op}));

  // Spine
  const spineBot = 50 - L*0.5;
  let d;
  if(L===0) d=`M${cx} ${headY+headR+1} Q${cx-7} ${headY+18} ${cx-5} ${headY+32} Q${cx-3} ${headY+43} ${cx-1} ${spineBot}`;
  else if(L===1) d=`M${cx} ${headY+headR+1} Q${cx-4} ${headY+16} ${cx-3} ${headY+30} L${cx-2} ${headY+44} L${cx} ${spineBot}`;
  else if(L===2) d=`M${cx} ${headY+headR+1} Q${cx-2} ${headY+14} ${cx-1} ${headY+28} L${cx} ${headY+43} L${cx+1} ${spineBot}`;
  else           d=`M${cx} ${headY+headR+1} L${cx} ${headY+26} L${cx} ${spineBot}`;
  svg.appendChild(mk('path',{d,stroke:'white','stroke-width':'3',fill:'none','stroke-linecap':'round',opacity:op}));

  // Arms
  const ay  = headY+headR+7+Math.max(0,(4-L)*1.5);
  const asx = 5+L*2.8, ady = 7-L*1.4;
  svg.appendChild(mk('line',{x1:cx,y1:ay,x2:Math.max(2,cx-asx).toFixed(1),y2:(ay+ady).toFixed(1),stroke:'white','stroke-width':'2.5','stroke-linecap':'round',opacity:op}));
  svg.appendChild(mk('line',{x1:cx,y1:ay,x2:Math.min(VW-2,cx+asx).toFixed(1),y2:(ay+ady).toFixed(1),stroke:'white','stroke-width':'2.5','stroke-linecap':'round',opacity:op}));

  // Legs
  const ly  = headY+headR+28-L*0.5;
  const lb  = Math.min(VH-2, headY+headR+48);
  const lsx = 5+L*1.0;
  svg.appendChild(mk('line',{x1:cx,y1:ly.toFixed(1),x2:Math.max(2,cx-lsx).toFixed(1),y2:lb,stroke:'white','stroke-width':'2.5','stroke-linecap':'round',opacity:op}));
  svg.appendChild(mk('line',{x1:cx,y1:ly.toFixed(1),x2:Math.min(VW-2,cx+lsx).toFixed(1),y2:lb,stroke:'white','stroke-width':'2.5','stroke-linecap':'round',opacity:op}));

  return svg;
}

    const silWrap = document.createElement('div');

    silWrap.appendChild(makeSilEl(a.level));

    // Text info
    const info = document.createElement('div');
    info.className='ach-info';
    info.innerHTML=`<div class="ach-day-lbl">День ${a.day}</div>
                    <div class="ach-name">${a.name}</div>
                    <div class="ach-desc">${a.desc}</div>`;

    // Check badge
    const chk = document.createElement('div');
    chk.className='ach-check';
    chk.textContent = ok?'✓':'';

    card.appendChild(silWrap);
    card.appendChild(info);
    card.appendChild(chk);
    shb.appendChild(card);
  });
}

function renderAI(d){
  const unlocked=ACHIEVEMENTS.filter(a=>a.day<=d).length;
  const pct=Math.round(Math.min(d/365,1)*100);
  const fc=FORECASTS.find(f=>d>=f.from&&d<=f.to)||FORECASTS[FORECASTS.length-1];
  const nextA=ACHIEVEMENTS.find(a=>a.day>d);

  document.getElementById('aiWrap').innerHTML=`
    <div class="ai-wrap">
      <div class="ai-row">
        <div class="ai-card"><div class="ai-cl">Streak</div><div class="ai-cv">${d}</div><div class="ai-cs">дней подряд</div></div>
        <div class="ai-card"><div class="ai-cl">Достижений</div><div class="ai-cv">${unlocked}/${ACHIEVEMENTS.length}</div><div class="ai-cs">разблокировано</div></div>
      </div>
      <div class="ai-bar-row"><span>До 365 дней</span><span>${pct}%</span></div>
      <div class="ai-track"><div class="ai-fill" style="width:${pct}%"></div></div>
      <div class="ai-fc">
        <div class="ai-fc-lbl">Прогноз</div>
        <div class="ai-fc-txt">${fc.txt}</div>
        <div class="ai-next">${nextA?'Следующее: «'+nextA.name+'» — день '+nextA.day+' (через '+(nextA.day-d)+' дн.)':'👑 Все достижения разблокированы'}</div>
      </div>
    </div>`;
}
